    function withTimeout(promise, ms = 45000) {
      return new Promise((resolve, reject) => {
        const t = setTimeout(() => reject(new Error('Request timed out. Click Repair or Connect.')), ms);
        promise.then(v => { clearTimeout(t); resolve(v); })
               .catch(e => { clearTimeout(t); reject(e); });

    function extractText(res) {
      if (res == null) return '';
      if (typeof res === 'string') return res;
      const c = res?.message?.content;
      if (Array.isArray(c)) {
        return c.filter(b => b?.type === 'text' && typeof b?.text === 'string')
                .map(b => b.text).join('\n').trim();
      }
      if (typeof c === 'string') return c;
      if (typeof res?.text === 'string') return res.text;
      if (typeof res?.content === 'string') return res.content;
      return safeStringify(res);
    }

    function splitArtifacts(text) {
      const out = [];
      const re = /```artifact:(\w+)\n([\s\S]*?)```/g;
      let last = 0, m;
      while ((m = re.exec(text)) !== null) {
        const before = text.slice(last, m.index);
        if (before.trim()) out.push({ kind:'text', value: before });
        out.push({ kind:'artifact', type: (m[1] || '').toLowerCase(), code: (m[2] || '').trim() });
        last = m.index + m[0].length;
      }
      const rest = text.slice(last);
      if (rest.trim()) out.push({ kind:'text', value: rest });
      return out.length ? out : [{ kind:'text', value: text }];
    }

    function renderText(el, text) {
      // Simple safe formatting: code fences + inline backticks.
      el.innerHTML = '';
      const parts = [];
      const re = /```(\w*)\n([\s\S]*?)```/g;
      let last = 0, m;
      while ((m = re.exec(text)) !== null) {
        const before = text.slice(last, m.index);
        if (before) parts.push({ type:'t', v: before });
        parts.push({ type:'c', lang: (m[1]||'').trim(), v: (m[2]||'').trim() });
        last = m.index + m[0].length;
      }
      const rest = text.slice(last);
      if (rest) parts.push({ type:'t', v: rest });

      for (const p of parts) {
        if (p.type === 't') {
          const div = document.createElement('div');
          let safe = escapeHtml(p.v);
          safe = safe.replace(/`([^`]+)`/g, '$1');
          safe = safe.replace(/\n/g, '
');
          div.innerHTML = safe;
          el.appendChild(div);
        } else {
          const pre = document.createElement('pre');
          const code = document.createElement('code');
          code.textContent = p.v;
          pre.appendChild(code);
          el.appendChild(pre);
        }
      }
    }

    async function fileToBase64Parts(file) {
      const dataUrl = await new Promise((resolve, reject) => {
        const r = new FileReader();
        r.onload = () => resolve(r.result);
        r.onerror = () => reject(new Error('Failed to read image file'));
        r.readAsDataURL(file);
      });

      const m = String(dataUrl).match(/^data:(image\/[a-zA-Z0-9.+-]+);base64,(.+)$/);
      if (!m) throw new Error('Invalid image encoding');
      return { media_type: m[1], data: m[2] };
    }

      hideBanner();

      // show user message immediately

      const imgUrls = state.pendingPreviewUrl ? [state.pendingPreviewUrl] : [];
      if (imgUrls.length) state.displayedUrls.push(imgUrls[0]); // keep for newChat cleanup

        state.pendingFile = null;
        state.pendingPreviewUrl = null;
        previewEl.classList.remove('active');
        previewEl.innerHTML = '';
        setMode('chat');

        const assistantEl = addMessage({ name:'Claude', avatar:'âœ¦', ai:true, text: '' });

        if (fileForThisRequest) {
          answer = await callVision(model, shownText, fileForThisRequest);
